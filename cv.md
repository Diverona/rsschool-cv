## Code example

function solve(arr) {
  let res = [];
  let alph = "abcdefghijklmnopqrstuvwxyz";

  for (let word of arr) {
    let sum = 0;
    let wordArr = word.toLowerCase().split("");
    for (let i = 0; i < wordArr.length; i++) {

    }
    res.push(sum);
  }

}
